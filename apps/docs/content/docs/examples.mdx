## Suspense streaming

The most common pattern: pass a promise to a component wrapped in `<Bones>`.

```tsx
import { Bones } from "bones";

async function fetchUser(): Promise<User> {
  const res = await fetch("/api/user");
  return res.json();
}

export default function Page() {
  return (
    <Bones>
      <UserCard user={fetchUser()} />
    </Bones>
  );
}

function UserCard({ user }: { user: Promise<User> }) {
  const { bone, data } = createBones(user);

  return (
    <div>
      <img {...bone("block")} src={data?.avatar} width={64} height={64} />
      <h2 {...bone("text", { length: 12 })}>{data?.name}</h2>
      <p {...bone("text", { lines: 2 })}>{data?.bio}</p>
    </div>
  );
}
```

## Forced skeletons for previews

Use `<BonesForce>` to show skeleton states without real data:

```tsx
import { BonesForce } from "bones";

function SkeletonDemo() {
  return (
    <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 16 }}>
      {/* Loaded state */}
      <UserCard user={Promise.resolve(mockUser)} />

      {/* Skeleton state */}
      <BonesForce>
        <UserCard />
      </BonesForce>
    </div>
  );
}
```

## Multi-line text

Use the `lines` option for paragraphs and descriptions:

```tsx
function ArticleCard({ article }: { article: Promise<Article> }) {
  const { bone, data } = createBones(article);

  return (
    <article>
      <h3 {...bone("text", { length: 20 })}>{data?.title}</h3>
      <p {...bone("text", { lines: 3 })}>{data?.excerpt}</p>
      <span {...bone("text", { length: 8, contained: true })}>
        {data?.readTime}
      </span>
    </article>
  );
}
```

The last line of a multi-line skeleton is automatically shorter (61% width) to look more natural.

## List placeholders with repeat

Use `repeat` to render a fixed number of placeholder items while data loads:

```tsx
function TagList({ pokemon }: { pokemon: Promise<Pokemon> }) {
  const { bone, data, repeat } = createBones(pokemon);

  return (
    <div>
      <h3 {...bone("text", { length: 9 })}>{data?.name}</h3>
      <div style={{ display: "flex", gap: 4 }}>
        {repeat(data?.types, 2).map((type, i) => (
          <span key={type ?? i} {...bone("text", { contained: true })}>
            {type}
          </span>
        ))}
      </div>
    </div>
  );
}
```

`repeat(data?.types, 2)` returns `[undefined, undefined]` while loading (rendering two skeleton pills), and the real `types` array once data arrives.
